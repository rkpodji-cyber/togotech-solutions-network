! ==============================================
! Configuration - SW2 (Switch succursale - TogoTech Solutions)
! Cisco Catalyst 2960-24TT
! ==============================================

hostname SW2
service password-encryption
enable secret cisco123

! --- VLAN unique de la succursale ---
vlan 60
 name Succursale
exit

! --- Ports d'accès pour les PC utilisateurs ---
interface range fastEthernet 0/1 - 4
 switchport mode access
 switchport access vlan 60
exit

interface fastEthernet 0/5
 switchport mode access
 switchport access vlan 60
exit

! --- Port vers R2 : doit appartenir au VLAN 60 pour relayer le trafic ---
! (correction appliquée après le premier test WAN qui avait échoué)
interface gigabitEthernet 0/1
 switchport mode access
 switchport access vlan 60
exit

end
write memory
